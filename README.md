### Basic use

```js
import LivePhotosKitReact from 'livephotoskit-react';

class App extends React.Component {
  render () {
    return (
      <div className="App">
        <LivePhotosKitReact
          photoSrc='./test.JPG'
          videoSrc='./test.mov'
        >
        </LivePhotosKitReact>
      </div>
    );
  }
}
```
